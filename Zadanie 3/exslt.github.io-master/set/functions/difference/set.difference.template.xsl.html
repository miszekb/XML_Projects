Source

<xsl:template name="set:difference">
   <xsl:param name="nodes1"
              select="/.." />
   <xsl:param name="nodes2"
              select="/.." />
   <xsl:apply-templates select="$nodes1[count(.|$nodes2) != count($nodes2)]"
                        mode="set:difference" />
</xsl:template>
<xsl:template match="node()|@*"
              mode="set:difference">
   <xsl:copy-of select="." />
</xsl:template>
